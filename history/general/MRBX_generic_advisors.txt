# For minor countries where advisors are not defined
# Generate them from a generic template

# Political advisors
every_possible_country = {
	limit = {
		has_country_flag = MRBX_ye_wanyong
	}
	generate_character = {
		token_base = generic_traitor # token will be TAG_token_base
		advisor = {
			slot = political_advisor
			traits = { MRBX_traitor }
			ai_will_do = {
				factor = 0
			}
		}
		portraits = { 
			army = { small = "GFX_portrait_KAYA" }
		}
	}
}
