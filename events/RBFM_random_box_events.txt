news_event = {
    id = RBFM_news_events.1
    title = RBFM_news_events.1.t
    desc = RBFM_news_events.1.desc
    picture = GFX_blue_event_001

    is_triggered_only = yes
    major = yes

    option = {
        name = RBFM_news_events.1.a
        trigger = {
            has_country_flag = RBFM_lucky_country
        }

        hidden_effect = {
            clr_country_flag = RBFM_lucky_country

		    random_country = {
			    limit = {
				    is_ai = no
			    }
			    set_country_flag = RBFM_lucky_country
		    }   
        }
    }

    option = {
        name = RBFM_news_events.1.b
        trigger = {
            NOT = {
                has_country_flag = RBFM_lucky_country
            }
        }
    }
}
